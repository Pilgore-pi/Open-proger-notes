
## База данных кетчупа

-- Create tables
CREATE TABLE Manufacturer (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    country VARCHAR(50) NOT NULL
);

CREATE TABLE Ketchup_Brand (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    manufacturer_id INTEGER REFERENCES Manufacturer(id),
    ketchup_speed INTEGER NOT NULL -- how fast is ketchup squeezed out of the bottle
);

CREATE TABLE Ingredient (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

CREATE TABLE Ketchup_Brand_Ingredient (
    ketchup_brand_id INTEGER REFERENCES Ketchup_Brand(id),
    ingredient_id INTEGER REFERENCES Ingredient(id),
    PRIMARY KEY (ketchup_brand_id, ingredient_id)
);

CREATE TABLE Store (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    city VARCHAR(50) NOT NULL
);

CREATE TABLE Store_Employee (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    store_id INTEGER REFERENCES Store(id),
    position VARCHAR(50) NOT NULL
);
