float4 Rectangle()
{
	float4 col;
	UV.x *= _Resolution;
	float2 WH = float2(0.5, 0.5); // Width, Height
	float2 shift = float2(0.45, 0.0);
	float2 bl = step((float2(1.0, 1.0) - WH) / 2 + shift, UV);
	float2 tr = step((float2(1.0, 1.0) - WH) / 2, 1.0-UV+shift);
	col = bl.x * bl.y * tr.x * tr.y;
	return col;
}

	float4 col;

	UV.x *= _Resolution;
	
	float comp =
		1 - step(0.2, distance(UV, float2(.5 + shift.x, .5)));
