```js
function showMessage(name) {
  alert('Привет, ' + `${name}!`);
}

// У функции есть доступ к внешней переменной
let userName = 'Вася';

function showMessage() {
  let message = 'Привет, ' + userName;
  alert(message);
}

// параметры по умлочанию
function Sum(a, b = 0) { return a + b; }

// возврат выражения, записанного в несколько строк
function getVal(...){
	return (
		some + long + expression
		+ or +
		whatever * f(a) + f(b)
	)
}
```

>Результат функции с пустым `return` или без него – `undefined`

В действительности при определении функции объявляется переменная функционального типа, то есть функция — это выражение

```js
let sayHi = function() {
  alert( "Привет" );
};
let theSame = sayHi;
theSame();
sayHi();
```

Передача функции в качестве параметра:

```js
function ask(question, yes, no) {
  if (confirm(question)) yes()
  else no();
}

ask(
  "Вы согласны?",
  function() { alert("Вы согласились."); },
  function() { alert("Вы отменили выполнение."); }
);
```

>Объявленная функция может быть вызвана когда угодно, в отличие от функции выражения, которую можно вызвать только после присваивания переменной этой функции.

## Лямбда-выражения

```js
let func = (arg1, arg2, ...argN) => expression;

// Аналогично
let func = function(arg1, arg2, ...argN) {
  return expression;
};

let sum = (a, b) => a + b;
let sayHi = () => alert("Hello!");

// Многострочная лямбда
let sum = (a, b) => {
  let result = a + b;
  return result;
};
```

> Лямбды не могут самостоятельно использовать ссылку this.
